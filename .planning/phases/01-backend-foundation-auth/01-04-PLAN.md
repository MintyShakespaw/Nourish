    ```typescript
    export async function forgotPassword(req: Request, res: Response): Promise<void> {
      const { email } = req.body as { email: string };

      // SECURITY: Always return 200 — no matter what we discover — to prevent enumeration.
      // Do the DB lookup unconditionally to keep timing roughly equal.
      const user = await userRepo.findByEmail(email);

      if (user) {
        const { raw, hash } = generatePasswordResetToken();
        await tokenRepo.createPasswordResetToken({
          userId: user.id,
          tokenHash: hash,
          expiresAt: passwordResetExpiry(),
        });
        const resetUrl = `${env.APP_BASE_URL}/reset-password?email=${encodeURIComponent(email)}&token=${raw}`;
        try {
          await sendPasswordResetEmail(email, resetUrl);
        } catch (err) {
          // Log but DO NOT surface — still respond 200.
          // eslint-disable-next-line no-console
          console.error('[forgotPassword] email send failed', err);
        }
      }

      res.status(200).json({ message: 'If that email is registered, a reset link has been sent.' });
    }

    export async function resetPassword(req: Request, res: Response): Promise<void> {
      const { email, token, newPassword } = req.body as {
        email: string;
        token: string;
        newPassword: string;
      };

      const user = await userRepo.findByEmail(email);
      if (!user) {
        res.status(400).json({ error: 'Invalid or expired reset token' });
        return;
      }

      // Find a matching unused, unexpired reset token via bcrypt.compare
      const candidates = await tokenRepo.findUnusedPasswordResetTokens(user.id);
      let matched: { id: string } | null = null;
      for (const row of candidates) {
        if (await bcrypt.compare(token, row.tokenHash)) {
          matched = { id: row.id };
          break;
        }
      }
      if (!matched) {
        res.status(400).json({ error: 'Invalid or expired reset token' });
        return;
      }

      const newHash = await bcrypt.hash(newPassword, 12);
      // Atomic-ish: update password + mark token used + revoke all active refresh tokens
      await tokenRepo.updateUserPassword(user.id, newHash);
      await tokenRepo.markPasswordResetTokenUsed(matched.id);
      await tokenRepo.revokeAllActiveByUser(user.id);

      res.status(200).json({ message: 'Password reset successful' });
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Public → `/api/auth/forgot-password` | Untrusted input; mitigated by always-200 response (enumeration) and no rate limit needed (no auth bypass possible from this endpoint) |
| Resend network call | Outbound dependency; failure must not leak existence of email |
| Reset-link URL → user's email inbox | The link contains the raw token in cleartext over the user's mail provider — assumed trusted enough for the threat model of a personal app |
| Public → `/api/auth/reset-password` | An attacker who steals the email link can change the password; mitigated by 1-hour expiry + single-use + revoking active sessions on success |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-24 | Information disclosure | Email enumeration via `forgot-password` | mitigate | Endpoint ALWAYS returns HTTP 200 with same body, regardless of whether the email is registered or Resend succeeded. DB lookup is performed even when no email will be sent (keeps timing roughly equal). |
| T-01-25 | Elevation of privilege | Reset token replay (D-04) | mitigate | `password_reset_tokens.used_at` is set on first successful use. `findUnusedPasswordResetTokens` filters `usedAt IS NULL AND expiresAt > NOW()`. Second attempt with same token finds no candidates → 400. |
| T-01-26 | Elevation of privilege | Reset token brute force | mitigate | Token is `crypto.randomBytes(32)` (256-bit), bcrypt-hashed at rest, 1-hour expiry. Search space for brute force on a 64-hex token is 2^256 — infeasible. |
| T-01-27 | Spoofing | Reset token used after password compromise to maintain access | mitigate | On successful reset, `revokeAllActiveByUser` revokes EVERY active refresh token for the user. Attacker's stolen refresh tokens become invalid immediately. |
| T-01-28 | Information disclosure | DB compromise reveals reset tokens | mitigate | Only the bcrypt hash is stored. Raw token exists only in the user's email and in the reset URL (the latter exists only between email-send and first click). |
| T-01-29 | Tampering | Race condition between mark-used and password update | accept | Three separate DB statements (`updateUserPassword`, `markPasswordResetTokenUsed`, `revokeAllActiveByUser`) are not wrapped in an explicit transaction. Worst-case race: a near-simultaneous second `resetPassword` call could complete between statements 1 and 2. Acceptable for single-user personal app; later phase could wrap in `db.transaction(...)` if needed. |
| T-01-30 | Denial of service | Email bombing via repeated `forgot-password` calls | accept | Resend's own per-account rate limit (100/day free tier) is the natural cap. No per-IP rate limit added in Phase 1 — acceptable trade-off for a personal-use app with no public sign-up. If abuse pattern emerges, add `forgotPasswordRateLimit` middleware mirroring `loginRateLimit`. |

## Residual Risk

- The reset link is sent over the user's email provider (Gmail / Outlook etc.). If the email provider is compromised, the attacker can complete the reset flow. This is the universal limit of email-based password recovery; acceptable.
- Reset URL appears in Express access logs (morgan). The token is high-entropy but logs should be protected. Documented in SKELETON.md "what this skeleton does not include" — production deployment would need log redaction.

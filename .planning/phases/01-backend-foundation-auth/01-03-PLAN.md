    import type { Request, Response, NextFunction } from 'express';
    import jwt from 'jsonwebtoken';
    import { verifyAccessToken } from '../services/auth.service.js';

    export function requireAuth(req: Request, res: Response, next: NextFunction): void {
      const header = req.headers.authorization;
      if (!header?.startsWith('Bearer ')) {
        res.status(401).json({ error: 'Missing or invalid Authorization header' });
        return;
      }
      try {
        const payload = verifyAccessToken(header.slice(7));
        res.locals.userId = payload.sub;
        next();
      } catch (err) {
        if (err instanceof jwt.TokenExpiredError || err instanceof jwt.JsonWebTokenError) {
          res.status(401).json({ error: 'Access token expired or invalid' });
          return;
        }
        res.status(401).json({ error: 'Access token expired or invalid' });
      }
    }

    import * as tokenRepo from '../repositories/token.repo.js';

    /**
     * Walks active refresh tokens for a user and finds the one whose bcrypt hash
     * matches the presented raw token. Returns the row (caller will rotate it) or null.
     */
    export async function verifyRefreshToken(
      presentedRaw: string,
      userId: string
    ): Promise<{ tokenRow: { id: string; userId: string; expiresAt: Date } } | null> {
      const active = await tokenRepo.findActiveByUser(userId);
      const now = new Date();
      for (const row of active) {
        if (row.expiresAt <= now) continue;
        const ok = await bcrypt.compare(presentedRaw, row.tokenHash);
        if (ok) return { tokenRow: row };
      }
      return null;
    }

    Append (keeping existing `register` handler intact):
    ```typescript
    import jwt from 'jsonwebtoken';
    import { env } from '../config/env.js';
    import {
      generateAccessToken,
      generateRefreshToken,
      refreshTokenExpiry,
      verifyRefreshToken as verifyRefreshTokenService,
    } from '../services/auth.service.js';

    export async function login(req: Request, res: Response): Promise<void> {
      const { email, password } = req.body as { email: string; password: string };

      const user = await userRepo.findByEmail(email);
      // Always run bcrypt to keep response timing consistent (prevents user-enumeration timing attack).
      // If user not found, compare against a fixed dummy hash to spend ~same time.
      const dummyHash = '$2b$12$abcdefghijklmnopqrstuv.0000000000000000000000000000000000.';
      const storedHash = user?.passwordHash ?? dummyHash;
      const ok = await bcrypt.compare(password, storedHash);

      if (!user || !ok) {
        res.status(401).json({ error: 'Invalid email or password' });
        return;
      }

      const accessToken = generateAccessToken(user.id);
      const { raw: refreshToken, hash: refreshHash } = generateRefreshToken();
      await tokenRepo.createRefreshToken({
        userId: user.id,
        tokenHash: refreshHash,
        expiresAt: refreshTokenExpiry(),
      });

      res.status(200).json({ accessToken, refreshToken });
    }

    export async function refresh(req: Request, res: Response): Promise<void> {
      const { refreshToken: presented } = req.body as { refreshToken: string };

      // We don't know the userId from the raw token alone — try every user that has any
      // active refresh token. In practice, refresh tokens are presented alongside an expired
      // access token (Axios interceptor flow); we accept that here we walk by user.
      // To find the user, the client SHOULD pass the access token (even expired) in body for hints.
      // Simpler approach used here: walk all active rows globally via a query.
      // Because verifyRefreshToken takes a userId, derive the userId by attempting to decode
      // the expired access token (without verifying) — this is information disclosure-safe
      // because the user must still present the matching refresh token.
      const accessHint = req.headers.authorization?.startsWith('Bearer ')
        ? req.headers.authorization.slice(7)
        : null;

      let userId: string | null = null;
      if (accessHint) {
        try {
          const decoded = jwt.decode(accessHint) as { sub?: string } | null;
          if (decoded?.sub) userId = decoded.sub;
        } catch {
          // ignore — fall through to 401
        }
      }
      if (!userId) {
        res.status(401).json({ error: 'Refresh token invalid' });
        return;
      }

      const match = await verifyRefreshTokenService(presented, userId);
      if (!match) {
        res.status(401).json({ error: 'Refresh token invalid' });
        return;
      }

      // Rotate: revoke the matched row and issue a new pair.
      await tokenRepo.revokeById(match.tokenRow.id);
      const newAccess = generateAccessToken(userId);
      const { raw: newRefresh, hash: newHash } = generateRefreshToken();
      await tokenRepo.createRefreshToken({
        userId,
        tokenHash: newHash,
        expiresAt: refreshTokenExpiry(),
      });

      res.status(200).json({ accessToken: newAccess, refreshToken: newRefresh });
    }

    export async function logout(req: Request, res: Response): Promise<void> {
      const userId = res.locals.userId as string;
      const { refreshToken: presented } = req.body as { refreshToken: string };

      const match = await verifyRefreshTokenService(presented, userId);
      if (match) {
        await tokenRepo.revokeById(match.tokenRow.id);
      }
      res.status(204).end();
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Public Internet → `/api/auth/login` | Untrusted brute-force attempts cross here — rate limiter is the perimeter |
| Refresh token rotation | A presented refresh token represents a captured session — must be invalidated on use |
| `Authorization: Bearer <token>` header | The sole authentication source for protected routes; never trust `req.body.userId` |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-17 | Spoofing | Login brute force | mitigate | `express-rate-limit`: 5 attempts per IP per 15 minutes returns 429. Production env never disables; test env skips for test isolation. |
| T-01-18 | Information disclosure | Email enumeration via login response | mitigate | Login ALWAYS runs `bcrypt.compare` (against a dummy hash when user missing) AND returns the identical error `'Invalid email or password'` for both unknown-email and wrong-password. Timing and response shape are uniform. |
| T-01-19 | Elevation of privilege | Refresh token replay after rotation | mitigate | Every successful `/refresh` calls `revokeById` on the matched row before issuing a new pair. Subsequent presentation of the same raw token finds no active row → 401. |
| T-01-20 | Spoofing | JWT algorithm confusion (alg:none) on `requireAuth` | mitigate | `verifyAccessToken` (Plan 01) pins `algorithms: ['HS256']`. `requireAuth` calls `verifyAccessToken` and rejects any verification failure. |
| T-01-21 | Information disclosure | Stolen refresh token used after device loss | mitigate | Refresh token is bcrypt-hashed in DB; raw value lives only in Expo SecureStore (hardware-backed). A second device successfully logging in does NOT invalidate other devices (D-02 30-day refresh) — explicit `/logout` from each device is the kill switch. Future enhancement (v2): "log out everywhere" via `revokeAllActiveByUser`. |
| T-01-22 | Tampering | Logout endpoint called for another user's refresh token | mitigate | Logout reads `userId` from `res.locals.userId` (set by `requireAuth` from the JWT). `verifyRefreshToken` scopes the search to that userId. An attacker with stolen access token cannot log out a different user — the refresh-token-to-user binding is enforced. |
| T-01-23 | Denial of service | Rate limiter exhausting memory at scale | accept | `express-rate-limit` default in-memory store is fine for a personal-use single-user app on a single Node process. Production hosted deployment (out of Phase 1 scope) would switch to a Redis store. |

## Residual Risk

- Refresh tokens have a 30-day lifetime (D-02). A stolen refresh token, if undetected, is usable for up to 30 days. Mitigation: rotation on every use + future "logout everywhere" feature.
- The login rate limit is per-IP. Behind a CGNAT / shared corporate IP, legitimate users could be limited together. Acceptable in personal-use MVP.
</threat_model>
